2、集合遍历

    public List<User> createList(){
        List<User> users = new ArrayList<User>();

        User user1 = new User();
        user1.setId(UUID.randomUUID().toString());
        user1.setName("李亦非");
        user1.setAge(20);
        user1.setSex(Sex.WOMAN);

        User user2 = new User();
        user2.setId(UUID.randomUUID().toString());
        user2.setName("张三");
        user2.setAge(20);
        user2.setSex(Sex.MAN);

        User user3 = new User();
        user3.setId(UUID.randomUUID().toString());
        user3.setName("刘三姐");
        user3.setAge(25);
        user3.setSex(Sex.WOMAN);

        users.add(user1);
        users.add(user2);
        users.add(user3);
        return users;
    }

    @Test
    public void lambTest(){
        List<User> users = createList();
        Stream<User> userStream = users.stream();
        //lambda表达式遍历集合
        userStream.forEach(user->{
            System.out.println("------"+user.toString());
        });

        //lambda过滤器
        userStream.filter(user->Sex.MAN.equals(user.getSex()))
                .forEach(user->{
            System.out.println(user.toString());
        });

        //lambda常用计算
        double asDouble = userStream.filter(u ->
                Sex.WOMAN.equals(u.getSex())
        ).mapToInt(user -> user.getAge()).average().getAsDouble();

        System.out.println(asDouble);
    }
